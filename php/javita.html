// Guardar las preguntas seleccionadas let pregunta1Select = document.querySelector('select[name="pregunta1"]'); let pregunta2Select = document.querySelector('select[name="pregunta2"]'); pregunta1Select.addEventListener('change', () => { // Al seleccionar una pregunta en el primer menú, deshabilitarla en el segundo menú let preguntaSeleccionada = pregunta1Select.value; pregunta2Select.querySelectorAll('option').forEach(option => { option.disabled = (option.value === preguntaSeleccionada); }); }); pregunta2Select.addEventListener('change', () => { // Al seleccionar una pregunta en el segundo menú, deshabilitarla en el primer menú let preguntaSeleccionada = pregunta2Select.value; pregunta1Select.querySelectorAll('option').forEach(option => { option.disabled = (option.value === preguntaSeleccionada); }); });
